#include <stdio.h>

int main() {
    int v[3];
    int m[3][3];
    int r[3];
    int i, j;

    // ler vetor
    printf("Digite o vetor:\n");
    for(i = 0; i < 3; i++) {
        scanf("%d", &v[i]);
    }

    // ler matriz
    printf("Digite a matriz 3x3:\n");
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            scanf("%d", &m[i][j]);
        }
    }

    // multiplicar vetor pelas colunas
    for(j = 0; j < 3; j++) {
        r[j] = 0;
        for(i = 0; i < 3; i++) {
            r[j] = r[j] + v[i] * m[i][j];
        }
    }

    // mostrar resultado
    printf("Resultado:\n");
    for(i = 0; i < 3; i++) {
        printf("%d ", r[i]);
    }

    return 0;
}
