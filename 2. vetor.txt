#include <stdio.h>

int main() {
    int vetor[5];
    int i;

    // Leitura dos 5 números
    for(i = 0; i < 5; i++) {
        printf("Digite o %dº número: ", i + 1);
        scanf("%d", &vetor[i]);
    }

    // Impressão em ordem inversa
    printf("\nVetor na ordem inversa:\n");
    for(i = 4; i >= 0; i--) {
        printf("%d ", vetor[i]);
    }

    return 0;
}
