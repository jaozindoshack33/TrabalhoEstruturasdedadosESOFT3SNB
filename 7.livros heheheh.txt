#include <stdio.h>
#include <string.h>

struct Livro {
    char titulo[30];
    char autor[15];
    int ano;
};

int main() {
    struct Livro livros[5];
    char busca[30];
    int i, achou = 0;

    // leitura dos dados
    for(i = 0; i < 5; i++) {
        printf("\nLivro %d\n", i + 1);

        printf("Titulo: ");
        scanf("%s", livros[i].titulo);

        printf("Autor: ");
        scanf("%s", livros[i].autor);

        printf("Ano: ");
        scanf("%d", &livros[i].ano);
    }

    // busca
    printf("\nDigite o titulo que deseja buscar: ");
    scanf("%s", busca);

    // procurar e mostrar
    printf("\nResultados:\n");
    for(i = 0; i < 5; i++) {
        if(strcmp(livros[i].titulo, busca) == 0) {
            printf("Titulo: %s | Autor: %s | Ano: %d\n",
                   livros[i].titulo,
                   livros[i].autor,
                   livros[i].ano);
            achou = 1;
        }
    }

    if(!achou) {
        printf("Nenhum livro encontrado.\n");
    }

    return 0;
}
